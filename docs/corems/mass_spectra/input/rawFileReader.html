 588    def get_average_mass_spectrum(
 589        self,
 590        spectrum_mode: str = "profile",
 591        auto_process: bool = True,
 592        ppm_tolerance: float = 5.0,
 593        ms_type: str = "MS1",
 594    ) -> MassSpecProfile | MassSpecCentroid:
 595        """
 596        Averages mass spectra over a scan range using Thermo's AverageScansInScanRange method
 597        or a scan list using Thermo's AverageScans method
 598        spectrum_mode: str
 599            centroid or profile mass spectrum
 600        auto_process: bool
 601            If true performs peak picking, and noise threshold calculation after creation of mass spectrum object
 602        ms_type: str
 603            String of form 'ms1' or 'ms2' or 'MS3' etc. Valid up to MS10.
 604            Internal function converts to Thermo MSOrderType class.
 605
 606        """
 607
 608        def get_profile_mass_spec(averageScan, d_params: dict, auto_process: bool):
 609            mz_list = list(averageScan.SegmentedScan.Positions)
 610            abund_list = list(averageScan.SegmentedScan.Intensities)
 611
 612            data_dict = {
 613                Labels.mz: mz_list,
 614                Labels.abundance: abund_list,
 615            }
 616
 617            return MassSpecProfile(data_dict, d_params, auto_process=auto_process)
 618
 619        def get_centroid_mass_spec(averageScan, d_params: dict):
 620            noise = list(averageScan.centroidScan.Noises)
 621
 622            baselines = list(averageScan.centroidScan.Baselines)
 623
 624            rp = list(averageScan.centroidScan.Resolutions)
 625
 626            magnitude = list(averageScan.centroidScan.Intensities)
 627
 628            mz = list(averageScan.centroidScan.Masses)
 629
 630            array_noise_std = (np.array(noise) - np.array(baselines)) / 3
 631            l_signal_to_noise = np.array(magnitude) / array_noise_std
 632
 633            d_params["baseline_noise"] = np.average(array_noise_std)
 634
 635            d_params["baseline_noise_std"] = np.std(array_noise_std)
 636
 637            data_dict = {
 638                Labels.mz: mz,
 639                Labels.abundance: magnitude,
 640                Labels.rp: rp,
 641                Labels.s2n: list(l_signal_to_noise),

596        """
597        Averages mass spectra over a scan range using Thermo's AverageScansInScanRange method
598        or a scan list using Thermo's AverageScans method
599        spectrum_mode: str
600            centroid or profile mass spectrum
601        auto_process: bool
602            If true performs peak picking, and noise threshold calculation after creation of mass spectrum object
603        ms_type: str
604            String of form 'ms1' or 'ms2' or 'MS3' etc. Valid up to MS10.
605            Internal function converts to Thermo MSOrderType class.
606
607        """
608
609        def get_profile_mass_spec(averageScan, d_params: dict, auto_process: bool):
610            mz_list = list(averageScan.SegmentedScan.Positions)
611            abund_list = list(averageScan.SegmentedScan.Intensities)
612
613            data_dict = {
614                Labels.mz: mz_list,
615                Labels.abundance: abund_list,
616            }
617
618            return MassSpecProfile(data_dict, d_params, auto_process=auto_process)
619

621            noise = list(averageScan.centroidScan.Noises)
622
623            baselines = list(averageScan.centroidScan.Baselines)
624
625            rp = list(averageScan.centroidScan.Resolutions)
626
627            magnitude = list(averageScan.centroidScan.Intensities)
628
629            mz = list(averageScan.centroidScan.Masses)
630
631            array_noise_std = (np.array(noise) - np.array(baselines)) / 3
632            l_signal_to_noise = np.array(magnitude) / array_noise_std
633
634            d_params["baseline_noise"] = np.average(array_noise_std)
635
636            d_params["baseline_noise_std"] = np.std(array_noise_std)
637
638            data_dict = {
639                Labels.mz: mz,
640                Labels.abundance: magnitude,
641                Labels.rp: rp,
642                Labels.s2n: list(l_signal_to_noise),
643            }
644
645            mass_spec = MassSpecCentroid(data_dict, d_params, auto_process=False)
646
647            return mass_spec

649        d_params = self.set_metadata(
650            firstScanNumber=self.start_scan, lastScanNumber=self.end_scan
651        )
652
653        # Create the mass options object that will be used when averaging the scans
654        options = MassOptions()
655        options.ToleranceUnits = ToleranceUnits.ppm
656        options.Tolerance = ppm_tolerance
657
658        # Get the scan filter for the first scan.  This scan filter will be used to located

691                raise ValueError("No data found for the selected scans")
692
693        elif isinstance(self.scans, list):
694            d_params = self.set_metadata(scans_list=self.scans)
695

697            for scan in self.scans:
