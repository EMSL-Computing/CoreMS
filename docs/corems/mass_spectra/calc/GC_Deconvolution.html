457    def quadratic_interpolation(self, rt_list, tic_list, apex_index):
458
459        rt_list = np.array(rt_list)
460        tic_list = np.array(tic_list)
461        three_highest_i = [i for i in range(apex_index - 1, apex_index + 2)]
462
463        z = np.poly1d(np.polyfit(rt_list[three_highest_i], tic_list[three_highest_i], 2))
464        a = z[2]
465        b = z[1]
466
467        corrected_apex_rt = -b / (2 * a)
468        initial_rt = rt_list[apex_index]
469
470        return initial_rt - corrected_apex_rt

460    def quadratic_interpolation(self, rt_list, tic_list, apex_index):
461
462        rt_list = np.array(rt_list)
463        tic_list = np.array(tic_list)
464        three_highest_i = [i for i in range(apex_index - 1, apex_index + 2)]
465
466        z = np.poly1d(np.polyfit(rt_list[three_highest_i], tic_list[three_highest_i], 2))
467        a = z[2]
468        b = z[1]
469
470        corrected_apex_rt = -b / (2 * a)
471        initial_rt = rt_list[apex_index]
472
473        return initial_rt - corrected_apex_rt

460    def quadratic_interpolation(self, rt_list, tic_list, apex_index):
461
