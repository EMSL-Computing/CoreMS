326def smooth_signal(x, window_len, window, pol_order, implemented_smooth_method):
327    
328    """smooth the data using a window with requested size.
329
330    This method is based on the convolution of a scaled window with the signal.
331    The signal is prepared by introducing reflected copies of the signal 
332    (with the window size) in both ends so that transient parts are minimized
333    in the begining and end part of the output signal.
334
335    output:
336        the smoothed signal
337        
338    see also: 
339
340    numpy.hanning, numpy.hamming, numpy.bartlett, numpy.blackman, numpy.convolve
341    scipy.signal.savgol_filter
342
343    """
344    x= np.array(x)
345
346    if x.ndim != 1:
347        raise ValueError("smooth only accepts 1 dimension arrays.")
348
349    if x.size < window_len:
350        raise ValueError("Input array needs to be bigger than window size")
351
352    #if window_len < 3:
353    #    return x
354
355    if not window in implemented_smooth_method:
356        raise ValueError("Window method should be 'flat', 'hanning', 'hamming', 'bartlett', 'blackman'")
357
358    s = np.r_[x[window_len - 1:0:-1], x, x[-1:-window_len:-1]]
359
360    if window == 'savgol':
361        
362        return savgol_filter(x, window_len, pol_order)
363
364
365    elif window == 'boxcar':  # moving average
366        
367        w = boxcar(window_len)
368        
369        y = np.convolve(w, s, mode='valid')
370
371        
372    elif window == 'flat':  # moving average
373        
374        w = np.ones(window_len, 'd')
375
376        y = np.convolve(w / w.sum(), s, mode='valid')
377        
378    else:
379        
380        w = eval(window + '(window_len)')
381
382        y = np.convolve(w / w.sum(), s, mode='valid')
383
384    return y[int(window_len / 2 - 1):int(-window_len / 2)]

386def second_derivative_threshold(tic, stds, rt, peak_height_max_percent, peak_max_prominence_percent):
