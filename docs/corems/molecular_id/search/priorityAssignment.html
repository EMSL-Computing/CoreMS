389    def get_classes_in_order(self, dict_ox_class_and_ms_peak)-> [(str, dict)]: 
390        """ Get the classes in order.
391        
392        Parameters
393        ----------
394        dict_ox_class_and_ms_peak : dict

396        
397        Returns
398        -------
399        list

402        Notes
403        -----
404        structure is 
405            ('HC', {'HC': 1})
406        """
407        
408        
409        usedAtoms = deepcopy(self.mass_spectrum_obj.molecular_search_settings.usedAtoms)
410        
411        usedAtoms.pop("C")
412        usedAtoms.pop("H")
413        usedAtoms.pop("O")
414
415        min_n, max_n = usedAtoms.get('N') if usedAtoms.get('N') else (0,0)
416        min_s, max_s = usedAtoms.get('S') if usedAtoms.get('S') else (0,0)
417        min_p, max_p = usedAtoms.get('P') if usedAtoms.get('P') else (0,0)
418
419        possible_n = [n for n in range(min_n, max_n + 1)]
420        possible_s = [s for s in range(min_s, max_s + 1)]
421        possible_p = [p for p in range(min_p, max_p + 1)]
422        
423        #used to enforce order for commum atoms 
424        # and track the atom index in on the tuple in all_atoms_tuples
425        atoms_in_order = ['N', 'S', 'P']
426        
427        #do number atoms prodcut and remove then from the usedAtoms dict
428        all_atoms_tuples = product(possible_n, possible_s, possible_p)
429        for atom in atoms_in_order:
430            
431            usedAtoms.pop(atom, None)
432        
433        #iterate over other atoms besides C,H, N, O, S and P
434        
435        for selected_atom_label, min_max_tuple in usedAtoms.items():
436            
437            min_x = min_max_tuple[0]
438            max_x = min_max_tuple[1]
439
440            possible_x = [x for x in range(min_x, max_x + 1)]
441            all_atoms_tuples = product(all_atoms_tuples, possible_x)
442            
443            #merge tuples
444            all_atoms_tuples = [all_atoms_combined[0] + (all_atoms_combined[1],) for all_atoms_combined in
445                                all_atoms_tuples]
446            
447            #add atom label to the atoms_in_order list
448            
449            #important to index where the atom position is in on the tuple in all_atoms_tuples

452        classes_strings_dict_tuples, hc_class = self.get_class_strings_dict(all_atoms_tuples, atoms_in_order)

454        combined_classes = self.combine_ox_class_with_other(atoms_in_order, classes_strings_dict_tuples, dict_ox_class_and_ms_peak)

458        oxygen_class_str_dict_tuple = [(ox_class, mspeak[0].class_dict) for ox_class, mspeak in dict_ox_class_and_ms_peak.items()]
