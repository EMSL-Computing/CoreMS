367    def init_engine(self, url):
368        """ Initializes the SQLAlchemy engine.
369
370        Parameters

373            The URL of the SQLite database.
374
375        Returns
376        --------

380        directory = os.getcwd()

431    def row_to_dict(self, row):
432        """ Converts a database row to a dictionary.
433        
434        Parameters
435        -----------
436        row : sqlalchemy.engine.row.Row
437            A row from the database.
438        
439        Returns
440        --------
441        dict
442            A dictionary representing the compound.
443        """
444        data_dict = {c.name: getattr(row, c.name) for c in row.__table__.columns}        
445
446        data_dict['mz'] = frombuffer(data_dict.get('mz'), dtype="int32")
447        data_dict['abundance'] = frombuffer(data_dict.get('abundance'), dtype="int32")
448
449        if row.metadatar:
450            data_dict['metadata'] = {c.name: getattr(row.metadatar, c.name) for c in row.metadatar.__table__.columns}
451
452        else:
453            data_dict['metadata'] = None
454
455        return data_dict

435        """ Converts a database row to a dictionary.
436        
437        Parameters
438        -----------
439        row : sqlalchemy.engine.row.Row
440            A row from the database.
441        
442        Returns
443        --------
444        dict
445            A dictionary representing the compound.
446        """
447        data_dict = {c.name: getattr(row, c.name) for c in row.__table__.columns}

450        data_dict['abundance'] = frombuffer(data_dict.get('abundance'), dtype="int32")
451
452        if row.metadatar:
453            data_dict['metadata'] = {c.name: getattr(row.metadatar, c.name) for c in row.metadatar.__table__.columns}
454
455        else:
456            data_dict['metadata'] = None
