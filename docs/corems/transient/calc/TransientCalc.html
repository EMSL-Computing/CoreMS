 72    def zero_fill(self, transient ):
 73        """ Zero fill the transient
 74        
 75        Parameters
 76        ----------
 77        transient : numpy.ndarray
 78            The transient data points
 79        
 80        Returns
 81        -------
 82        numpy.ndarray
 83            The transient data points zerofilled
 84            
 85        Notes
 86        -----
 87        The number of zero fills is defined by the transient parameter number_of_zero_fills.
 88        The function first calculate the next power of two of the transient length and zero fills to that length, to take advantage of FFT algorithm.
 89            If the parameter next_power_of_two is set to False, the function will zero fill to the length of the original transient times the number of zero fills
 90
 91        """
 92        if self.parameters.next_power_of_two:
 93            exponent = int(ceil(log2(len(transient)*(self.parameters.number_of_zero_fills+1))))
 94            zeros_filled_transient = zeros(2**exponent)
 95        else:
 96            zeros_filled_transient = zeros(len(transient)*(self.parameters.number_of_zero_fills+1))
 97                
 98        zeros_filled_transient[0:len(transient)] = transient    
 99        
100        del transient
101        
102        gc.collect()
103        
104        return  zeros_filled_transient

138    def apodization(self, transient):
139        """ Apodization of the transient
140
141        Parameters
142        ----------
143        transient : numpy.ndarray
144            The transient data points
145        
146        Returns
147        -------
148        numpy.ndarray
149            The apodized transient data points
150        
151        Notes
152        -----
153        The apodization method is defined by the transient parameter apodization_method.
154        The following apodization methods are available:
155            Hamming,
156            Hanning,
157            Blackman,
158            Full-Sine,
159            Half-Sine,
160            Kaiser,
161            Half-Kaiser.
162        
163        For Kaiser and Half-Kaiser, an additional parameter 'beta' is required, set by the transient parameter kaiser_beta.
164
165        """
166        
167        apodi_method = self.parameters.apodization_method
168        beta = self.parameters.kaiser_beta
169        
170        length = len(transient)
171            
172        if apodi_method == "Hamming":
173                H_function = hamming(length)
174        elif apodi_method == "Hanning":
175                H_function = hanning(length)
176        elif apodi_method == "Blackman":
177                H_function = blackman(length)
178        elif apodi_method == "Full-Sine":
179                H_function = sin(linspace(0,pi,num=length))
180        elif apodi_method == "Half-Sine":
181                H_function = sin(linspace((pi/2),0,num=length))
182        elif apodi_method == "Kaiser":
183                H_function = kaiser(length,beta)
184        elif apodi_method == "Half-Kaiser":
185                H_function = kaiser(length*2,beta)[length:]
186            
187        S_x = transient * H_function
188        
189        del transient
190        gc.collect()  
191            
192        return S_x
