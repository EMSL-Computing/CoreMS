289    def mformula_s_to_dict(self, s_mformulatring, iontype='unknown'):
290        """Converts a molecular formula string to a dict
291        
292        Parameters
293        ----------
294        s_mformulatring : str
295            molecular formula string, i.e. 'C10H21NNa'
296        iontype : str, optional
297            ion type. Default is 'unknown'
298        
299        Returns
300        -------
301        dict
302            molecular formula dictionary
303
304        Notes
305        -----
306        Does not work if the atomic mass number is passed i.e. 37Cl, 81Br, convention follow the light isotope labeling 35Cl is Cl, 12C is C, etc.
307        If you need to use heavy isotopes please use another reference file format that separate the formula string by a blank space and parse it using the function corems_ref_file
308
309        Raises
310        ------
311        TypeError
312            Atom does not exist in Atoms.atoms_order list
313        Exception
314            Empty molecular formula
315        """
316        if s_mformulatring:
317            
318            #find the case C122
319            all_atoms = re.findall(r'[A-Z]{1}[0-9]{1,10000}', s_mformulatring)
320            
321            #find the case Br2
322            all_atoms2 = re.findall(r'[A-Z]{1}[a-z]{1}[0-9]{1,10000}', s_mformulatring)
323            #find the case N
324            single_digit_atoms_one = re.findall(r'[A-Z]{1}(?![0-9])(?![a-z])', s_mformulatring)
325            #print(single_digit_atoms_one)
326            #find the case Na
327            due_digit_atoms_one = re.findall(r'[A-Z]{1}[a-z]{1}(?![0-9])', s_mformulatring)
328            
329            all_atoms = all_atoms + all_atoms2 + due_digit_atoms_one +single_digit_atoms_one
330            
331            dict_res = {}
332            
333            for each_atom_count in all_atoms:
334                
335                
336                count = re.findall(r'[0-9]{1,10000}', each_atom_count)
337                atom = ''.join(re.findall(r'[A-z]', each_atom_count))
338                
339                if atom in Atoms.atoms_order:
340                    
341                    if count:
342                        dict_res[atom] = int(count[0])
343                    else:
344                        dict_res[atom] = 1
345                
346                else:
347                    
348                    tb = sys.exc_info()[2]
349                    raise TypeError("Atom %s does not exist in Atoms.atoms_order list" % atom).with_traceback(tb)
350            
351            dict_res[Labels.ion_type]  = iontype
352
353            return dict_res
354        
355        else: 
356            
357            tb = sys.exc_info()[2]
358            raise Exception('Empty molecular formula').with_traceback(tb)
