 78    def extrapolate_axes_for_pp(self):
 79        """ Extrapolate the m/z axis and fill the abundance axis with 0s.
 80
 81        Returns
 82        -------
 83        mz : ndarray
 84            The extrapolated m/z axis.
 85        abund : ndarray
 86            The abundance axis with 0s filled.
 87        freq : ndarray or None
 88            The extrapolated frequency axis, if available.
 89
 90        Notes
 91        --------
 92        This function will extrapolate the mz axis by N datapoints, and fill the abundance axis with 0s. 
 93        This should prevent peak picking issues at the spectrum edge.
 94
 95        """
 96        
 97        
 98        def extrapolate_axis(initial_array, pts):
 99           """
100           this function will extrapolate an input array in both directions by N pts
101           """
102           initial_array_len = len(initial_array)
103           right_delta = initial_array[-1] - initial_array[-2]  
104           left_delta = initial_array[1] - initial_array[0]  
105
106           pad_array = pad(initial_array, (pts, pts), 'constant', constant_values= (0,0))
107           ptlist = list(range(pts))
108           for pt in range(pts+1):
109               # This loop will extrapolate the right side of the array
110               final_value = initial_array[-1]
111               value_to_add = (right_delta*pt)
112               new_value = final_value+value_to_add
113               index_to_update = initial_array_len+(pt+1)
114               pad_array[index_to_update] = new_value
115               
116           for pt in range(pts+1):
117               # this loop will extrapolate the left side of the array
118               first_value = initial_array[0]
119               value_to_subtract = (left_delta*pt)
120               new_value = first_value-value_to_subtract
121               pad_array[ptlist[-pt]] = new_value
122               
123           return pad_array 
124        
125        mz, abund = self.mz_exp_profile, self.abundance_profile
126        if self.has_frequency:
127            freq = self.freq_exp_profile
128        else: freq = None
129        pts = self.settings.picking_point_extrapolate
130        if pts == 0:
131            return mz, abund, freq
132        
133        mz = extrapolate_axis(mz, pts)
134        abund = pad(abund, (pts, pts), mode = 'constant', constant_values=(0,0))
135        if freq is not None:
136                freq = extrapolate_axis(freq, pts)
137        return mz, abund, freq

173    def find_minima(self, apex_index, abundance, len_abundance, right=True):
174        """ Find the minima of a peak.
175
176        Parameters
177        ----------
178        apex_index : int
179            The index of the peak apex.
180        abundance : ndarray
181            The abundance values.
182        len_abundance : int
183            The length of the abundance array.
184        right : bool, optional
185            Flag indicating whether to search for minima to the right of the apex (default is True).
186
187        Returns
188        -------
189        int
190            The index of the minima.
191
192        """
193        j = apex_index
194        
195        if right: minima = abundance[j] > abundance[j+1]
196        else: minima = abundance[j] > abundance[j-1]
197
198        while minima:
199            
200            if j == 1 or j == len_abundance -2:
201                break
202            
203            if right: 
204                j += 1
205
206                minima = abundance[j] >= abundance[j+1]
207
208            else: 
209                j -= 1
210                minima = abundance[j] >= abundance[j-1]
211        
212        if right: return j
213        else: return j

215    def calculate_resolving_power(self, intes, massa, current_index):
216        """ Calculate the resolving power of a peak.
217
218        Parameters
219        ----------
220        intes : ndarray
221            The intensity values.
222        massa : ndarray
223            The mass values.
224        current_index : int
225            The index of the current peak.
226
227        Returns
228        -------
229        float
230            The resolving power of the peak.
231
232        Notes
233        --------
234        This is a conservative calculation of resolving power,
235        the peak need to be resolved at least at the half-maximum magnitude,
236        otherwise, the combined full width at half maximum is used to calculate resolving power.
237
238        """
239
240        peak_height = intes[current_index]
241        target_peak_height = peak_height/2
242
243        peak_height_minus = peak_height
244        peak_height_plus = peak_height
245        
246        # There are issues when a peak is at the high or low limit of a spectrum in finding its local minima and maxima
247        # This solution will return nan for resolving power when a peak is possibly too close to an edge to avoid the issue
248        
249        if current_index <5:
250            print("peak at low spectrum edge, returning no resolving power")
251            return nan
252        elif abs(current_index-len(intes))<5:
253            print("peak at high spectrum edge, returning no resolving power")
254            return nan
255        else:
256            pass
257
258        index_minus = current_index
259        while peak_height_minus  >= target_peak_height:
260
261            index_minus = index_minus -1
262            
263            # TODO : see if this try/except can be removed.
264            
265            try: 
266                peak_height_minus = intes[index_minus]
267            except IndexError:
268                print('Res. calc. warning - peak index minus adjacent to spectrum edge \n \
269                        Zeroing the first 5 data points of abundance. Peaks at spectrum edge may be incorrectly reported')
270                intes[:5] = 0
271                peak_height_minus = target_peak_height
272                index_minus -= 1 
273            #print "massa", "," , "intes", "," , massa[index_minus], "," , peak_height_minus
274        x = [ massa[index_minus],  massa[index_minus+1]]
275        y = [ intes[index_minus],  intes[index_minus+1]]
276        coefficients = polyfit(x, y, 1)
277
278        a = coefficients[0]
279        b = coefficients[1]
280        if self.mspeaks_settings.legacy_resolving_power:
281            y_intercept =  intes[index_minus] + ((intes[index_minus+1] - intes[index_minus])/2)
282        else:
283            y_intercept =  target_peak_height
284        massa1 = (y_intercept -b)/a
285
286        index_plus = current_index
287        while peak_height_plus  >= target_peak_height:
288
289            index_plus = index_plus + 1
290            
291            # TODO : see if this try/except can be removed.
292            
293            try: 
294                peak_height_plus = intes[index_plus]
295            except IndexError:
296                print('Res. calc. warning - peak index plus adjacent to spectrum edge \n \
297                        Zeroing the last 5 data points of abundance. Peaks at spectrum edge may be incorrectly reported')
298                intes[-5:] = 0
299                peak_height_plus = target_peak_height
300                index_plus += 1 
301            #print "massa", "," , "intes", "," , massa[index_plus], "," , peak_height_plus
302
303        x = [massa[index_plus],  massa[index_plus - 1]]
304        y = [intes[index_plus],  intes[index_plus - 1]]
305
306        coefficients = polyfit(x, y, 1)
307        a = coefficients[0]
308        b = coefficients[1]
309
310        if self.mspeaks_settings.legacy_resolving_power:
311            y_intercept =  intes[index_plus - 1] + ((intes[index_plus] - intes[index_plus - 1])/2)
312        else:
313            y_intercept =  target_peak_height
314
315        massa2 = (y_intercept -b)/a
316
317        if massa1 > massa2:
318
319            resolvingpower =  massa[current_index]/(massa1-massa2)
320
321        else:
322
323            resolvingpower =  massa[current_index]/(massa2-massa1)
324
325        return resolvingpower

100           """
101           this function will extrapolate an input array in both directions by N pts
102           """

104           right_delta = initial_array[-1] - initial_array[-2]  
105           left_delta = initial_array[1] - initial_array[0]

108           ptlist = list(range(pts))

110               # This loop will extrapolate the right side of the array
111               final_value = initial_array[-1]
112               value_to_add = (right_delta*pt)
113               new_value = final_value+value_to_add
114               index_to_update = initial_array_len+(pt+1)
115               pad_array[index_to_update] = new_value
116               
117           for pt in range(pts+1):
118               # this loop will extrapolate the left side of the array
119               first_value = initial_array[0]
120               value_to_subtract = (left_delta*pt)
121               new_value = first_value-value_to_subtract
122               pad_array[ptlist[-pt]] = new_value
123

252            return nan
